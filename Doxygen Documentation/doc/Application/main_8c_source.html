55static void Movement_handler(void*parameters)
56{
57 s8 x=0;
58 u32 speed=0;
59 while(1)
60 {
61 x=0;
62 MUART2_u8ReceiveWithoutBlocking(&x);
63 switch(x)
64 {
65 case 'F':
66 HMotor_VoidCarForword();
67 break;
68 case 'B':
69 HMotor_VoidCarBackword();
70 break;
71 case 'R':
72 HMotor_VoidCarRight();
73 break;
74 case 'L':
75 HMotor_VoidCarLeft();
76 break;
77 case 'S':
78 speed=speed+100;
79 HMotor_VoidCarSpeed(speed);
80 break;
81 case 'H':
82 speed=speed-100;
83 HMotor_VoidCarSpeed(speed);
84 break;
85 case 'Y':
86 break;
87 default :
88 HMotor_VoidCarStop();
89 break;
90 }
91
92 vTaskDelay(1);
93 }
94}

102static void Ultrasonic_handler(void*parameters)
